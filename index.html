Задача №1

Написать, функцию, которая принимает в качестве аргумента объект и выводит в консоль все ключи и значения только собственных свойств. Данная функция не должна возвращать значение.

Решение:

const userNew = {
Имя: 'Дмитрий',
Фамилия: 'Анисимов',
Возраст: '41'
} Object.getOwnPropertyNames(userNew).forEach(function(ident) {
console.log(ident + ': ' + userNew[ident]);
});

Задача №2

Написать функцию, которая принимает в качестве аргументов строку и объект, а затем проверяет есть ли у переданного объекта свойство с данным именем. Функция должна возвращать true или false.

Решение:

function showDetails(obj, objKeyName) {
return (objKeyName in obj);
}
const userNew = {
name: "Дмитрий",
ownCity: "Казань",
age: 41,
course: "Frontend"
};
console.log(showDetails(userNew, "surname"));
console.log(showDetails(userNew, "name"));

Задача №3

Написать функцию, которая создает пустой объект, но без прототипа.

Решение:

function createEmptyObject () {
return Object.create(null);
}
const obj = createEmptyObject();
console.log(obj);
console.log(Object.getPrototypeOf(obj));

Реализуйте следующее консольное приложение подобно примеру, который разбирался в видео. Реализуйте его на прототипах. Определите иерархию электроприборов. Включите некоторые в розетку. Выбрав прибор, подумайте, какими свойствами он обладает.

Решение:

function ElectricalAppliance(name) {
this.device = ' Electrical device',
this.name = name
}
ElectricalAppliance.prototype.getPowerExpense = function (power, time, on) {
let turnOn = on;
if(on === 1){
turnOn = 'Вкл';
}else{
turnOn = 'Выкл';
};
console.log(`Расход эл. энергии ${this.name} за ${time} час. = ${power * time}Вт. ${this.name}: ${turnOn}`);};
function Specification(name, color) {
this.name = name,
this.color = color
}

const kettle = new Specification ('Чайник', 'Белый');

Задача №5

Переписать консольное приложение из предыдущего юнита на классы.

Решение:

class ElectricalAppliance {
constructor(name) {
this.device = 'Electrical device';
this.name = name;
}
getPowerExpense (power, time, on) {
let turnOn = on;
if(on === 1){
turnOn = 'Вкл';
}else{
turnOn = 'Выкл';
};
console.log(`Расход эл. энергии ${this.name} - (${this.color}) за ${time} час. = ${power * time}Вт. ${this.name}: ${turnOn}`);
}
}
class Specification extends ElectricalAppliance {
constructor (name, color) {
super(name);
this.color = color;
}
}
const kettle = new Specification ('Чайник', 'Белый');
const laptop = new Specification ('Ноутбук', 'Чёрный');
const lamp = new Specification ('Лампочка', 'Красный');
console.log(kettle, laptop, lamp);
kettle.getPowerExpense(200, 2);
laptop.getPowerExpense(100, 2, 1);
lamp.getPowerExpense(50, 2, 1);
